git init                    --> Create a new local repository
git clone <url>             --> Download from an existing repository
git status                  --> List new or modified files not yet committed
git add <fil-name>          --> Stages the file and ready for commit
git commit -m <detail>      --> Commit all staged files to versioned history
git push                    --> Push local changes to origin
git pull                    --> Fetch the latest changes from origin and merge
git log                     --> Show full change history
git branch                  --> List all local branches
git branch <branch-name>    --> Create a new branch
git checkout <branch-name>  --> Switch to a branch
git reset <file-name>       --> Unstages the file, keeping the file changes
git diff                    --> Show the changes to files not yet staged
